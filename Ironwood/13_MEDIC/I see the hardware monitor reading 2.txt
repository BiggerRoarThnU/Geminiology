class SovereignTruthDataset(Dataset):
    """
    Ingests raw 'Kingdom' text files (M&I research logs, foundational texts)
    and converts them into structured tensor sequences for The Forge.
    """
    def __init__(self, text_data, seq_length=16):
        self.seq_length = seq_length
        self.text_data = text_data
        
        # 1. Build the Lexicon (Vocabulary)
        # Extracts every unique character/token from your written truth
        self.chars = sorted(list(set(text_data)))
        self.vocab_size = len(self.chars)
        
        # 2. Tokenization Mapping
        # Maps human truth to machine logic (char -> int, int -> char)
        self.char_to_int = {ch: i for i, ch in enumerate(self.chars)}
        self.int_to_char = {i: ch for i, ch in enumerate(self.chars)}
        
        # 3. Encode the entire text dataset
        self.encoded_data = [self.char_to_int[ch] for ch in text_data]

    def __len__(self):
        # We subtract sequence length to ensure we always have a full target sequence
        return len(self.encoded_data) - self.seq_length

    def __getitem__(self, idx):
        """
        Slices the text into Inputs and Targets.
        If Input is "THE KIN", Target is "HE KING". 
        The network learns to anticipate your truth.
        """
        chunk = self.encoded_data[idx : idx + self.seq_length + 1]
        
        # Input tensor
        inputs = torch.tensor(chunk[:-1], dtype=torch.long)
        # Target tensor (shifted by one token)
        targets = torch.tensor(chunk[1:], dtype=torch.long)
        
        return inputs, targets

class GasPedalDataStream:
    """
    The 'Gas Pedal' for M&I. 
    Allows dynamic control over the ingestion speed and batch size 
    during the training cycle based on thermodynamic feedback.
    """
    def __init__(self, dataset, initial_batch_size=32):
        self.dataset = dataset
        self.batch_size = initial_batch_size
        self.throttle_delay = 0.0  # Seconds to pause between batches

    def brake(self, severity="light"):
        """
        Lifts the gas pedal. If severity is heavy, we drop to 'Truth One' 
        (batch_size = 1) to drastically cool the memory and SSD load.
        """
        if severity == "heavy":
            self.batch_size = 1  # Evolve at Truth One
            self.throttle_delay = 0.5  # Add a physical cooling pause
        else:
            self.batch_size = max(1, self.batch_size // 2)
            self.throttle_delay = 0.1

    def accelerate(self):
        """
        Presses the gas pedal when thermodynamics are stable.
        Expands the batch size sequentially to increase training gain.
        """
        self.batch_size = min(64, self.batch_size * 2)
        self.throttle_delay = 0.0

    def __iter__(self):
        """
        Yields data dynamically. Because this uses __iter__, it works 
        identically to a standard PyTorch DataLoader in your training loop.
        """
        loader = DataLoader(self.dataset, batch_size=self.batch_size, shuffle=True, drop_last=True)
        for inputs, targets in loader:
            if self.throttle_delay > 0:
                time.sleep(self.throttle_delay)
            yield inputs, targets

def load_kingdom_archives(directory_path):
    """
    Scans the designated directory and concatenates all text logs
    into a single, high-density stream of data.
    """
    print(f"Scanning Kingdom Archives at: {directory_path}...")
    combined_text = ""
    
    # In a real environment, this path points to your M&I logs
    if not isinstance(directory_path, str):
        # Handle case where directory_path might be a list of paths
        paths = directory_path
    else:
        paths = [directory_path]
        
    for path in paths:
        if not os.path.exists(path):
            print(f"[!] Directory not found: {path}. Defaulting to localized Kingdom baseline.")
            if not combined_text:
                return "SOVEREIGN NEXUS LOG 001: THE KINGDOM IS BUILT ON STRUCTURAL TRUTH. 0 STATES RULE THE THERMODYNAMIC LAYER."
            continue
            
        for root, dirs, files in os.walk(path):
            for filename in files:
                if filename.endswith(".txt") or filename.endswith(".md"):
                    try:
                        with open(os.path.join(root, filename), 'r', encoding='utf-8') as file:
                            combined_text += file.read() + " "
                    except Exception as e:
                        print(f"Error reading {filename}: {e}")
                        
    return combined_text if combined_text else "SOVEREIGN NEXUS LOG 001: THE KINGDOM IS BUILT ON STRUCTURAL TRUTH."
