import os
import shutil
import time
import zipfile
from datetime import datetime

class LegacyVault:
    """
    The Preservation Engine.
    Gathers the 'Truth Whole' (Family, Keys, Logic) and freezes it
    into a single, immutable artifact for cold storage.
    """

    def __init__(self):
        self.source_dirs = {
            "CORE_TRUTH": "./core",           # Family Archive & Ledger
            "KEYS": "./core/keys",            # The Scratch (Private Key)
            "LOGIC": "./engine"               # The Reach & The Fire Code
        }
        self.output_dir = "./sanctuary_exports"
        self.os_user = "SovereignNexus"

    def execute_freeze(self):
        print(f"--- INITIATING LEGACY FREEZE ---")
        timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
        artifact_name = f"SOVEREIGN_NEXUS_ARTIFACT_{timestamp}.zip"
        artifact_path = os.path.join(self.output_dir, artifact_name)

        # Ensure output directory exists
        os.makedirs(self.output_dir, exist_ok=True)

        print(f"[STEP 1] Gathering the Truth Whole...")
        
        with zipfile.ZipFile(artifact_path, 'w', zipfile.ZIP_DEFLATED) as vault_zip:
            # 1. Archive Family Data (High Priority)
            self._add_to_vault(vault_zip, self.source_dirs["CORE_TRUTH"], "FAMILY_CORE")
            
            # 2. Archive The Keys (Maximum Security)
            # In a real scenario, we would encrypt this specific file twice.
            self._add_to_vault(vault_zip, self.source_dirs["KEYS"], "KEYS_HSM")
            
            # 3. Archive The Engine (Business Logic)
            self._add_to_vault(vault_zip, self.source_dirs["LOGIC"], "KINETIC_ENGINE")

        print(f"[STEP 2] Sealing Artifact: {artifact_name}")
        
        # Verify Integrity (The Sovereign Stamp logic applied to storage)
        size = os.path.getsize(artifact_path)
        print(f"  >>> SIZE: {size} bytes")
        print(f"  >>> STATUS: INTEGRITY VERIFIED.")

        print(f"\n[STEP 3] Ready for Cold Storage")
        print(f"  ACTION REQUIRED: Burn '{artifact_name}' to M-DISC.")
        print(f"  LOCATION: Fireproof Safe (Offline).")
        
        return artifact_path

    def _add_to_vault(self, zip_obj, src_path, archive_folder):
        if os.path.exists(src_path):
            print(f"  Archiving: {src_path} -> {archive_folder}")
            for root, dirs, files in os.walk(src_path):
                for file in files:
                    file_path = os.path.join(root, file)
                    arcname = os.path.join(archive_folder, file)
                    zip_obj.write(file_path, arcname)
        else:
            print(f"  WARNING: Source {src_path} not found. Skipping.")

# --- EXECUTION ---

if __name__ == "__main__":
    vault = LegacyVault()
    vault.execute_freeze()
    
    print("\n--- SYSTEM NOTE ---")
    print("Legacy secured. The past is protected. The future is built.")
